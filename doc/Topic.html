def initialize(name, time = nil, history = nil)
  @name = name
  @history = Array.new
  @history.push(history) unless history.nil?
  @last_updated = time.nil? ? Time.now : time 
  @count = @history.count

def read(options = {})
  
  messages = Array.new()
  
  limit  = options[:limit]  || nil
  
  if limit.nil?
    # All of it
    return @history
  else
    # Some of it
    @history.each do |item|
      limit -= 1 unless limit.nil?
      messages.push(item)
      break if limit == 0
    end
    return messages
  end
end
